A-Frame

可以在網頁中直接顯示 VR 體驗。

A-Frame官網

範例程式碼

<script src="https://aframe.io/releases/1.6.0/aframe.min.js"></script>

            <a-scene>
                <a-light type="ambient" color="#ffffff" intensity="0.5"></a-light>
                <a-light type="point" position="2 4 -4" intensity="1.5" color="#FFF"></a-light>
                <a-box position="-1 0.5 -3" rotation="0 45 0" color="#4CC3D9" 
                        animation="property: rotation; to: 0 405 0; dur: 3000; easing: linear; loop: true"></a-box>
                <a-sphere position="1 1 -5" radius="1.25" color="#EF2D5E"
                           animation="property: position; to: 1 2.5 -5; dur: 1500; direction: alternate; loop: true"></a-sphere>
                <a-cylinder position="0 0 -5" radius="0.5" height="1.5" color="#FFC65D"
                             animation="property: color; to: #FF6347; dur: 2000; direction: alternate; loop: true"></a-cylinder>
                <a-plane position="0 0 -4" rotation="-90 0 0" width="10" height="10" color="#7BC8A4"></a-plane>
                <a-sky color="#ECECEC"></a-sky>
                <a-camera position="0 1.6 3">
                    <a-cursor color="#FF0000"></a-cursor>
                </a-camera>
            </a-scene>
